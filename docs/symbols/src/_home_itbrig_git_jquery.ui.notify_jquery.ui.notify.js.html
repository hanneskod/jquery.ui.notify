 10 /**

 17 
 18 (function($){
 19 
 20     /**
 21      * @desc Growl like notifications styled with themeroller
 22      * @name jQuery.notify
 23      * @function
 24      * @param {obj} options
 25      * <dl>
 26      *  <dt>inEffect</dt>
 27      *  <dd>In effect</dd>
 28      *  <dt>outEffect</dt>
 29      *  <dd>Out effect</dd>
 30      *  <dt>effectTime</dt>
 31      *  <dd>Effect duration in miliseconds</dd>
 32      *  <dt>timeout</dt>
 33      *  <dd>Timeout in miliseconds before item is removed</dd>
 34      *  <dt>text</dt>
 35      *  <dd>Item text</dd>
 36      *  <dt>remove</dt>
 37      *  <dd>Should item be removed after timeout</dd>
 38      *  <dt>state</dt>
 39      *  <dd>ui-state- class</dd>
 40      *  <dt>icon</dt>
 41      *  <dd>ui-icon- class</dd>
 42      * </dl>
 43      * @see example.html
 44      */
 45     $.fn.notify = function(options){
 46         return this.each(function(){
 47             var $this = $(this);
 48  
 49             // Apply standard settings
 50             options = $.extend({
 51                 inEffect:   {opacity: 'show'},      // In effect
 52                 outEffect:  {opacity: '0'},         // Out effect
 53                 effectTime: 100,                    // Effect duration in miliseconds
 54                 timeout:    10000,                  // Timeout in miliseconds before item is removed
 55                 text:       '',                     // Item text
 56                 remove:     true,                   // Should item be removed after timeout
 57                 state:      'ui-state-highlight',   // jquery-ui state (anything that makes a valid ui-state- class
 58                 icon:       'ui-icon-info',         // jquery-ui icon (anything that makes a vlid ui-icon- class
 59             }, options);
 60 
 61             // Create item
 62             var $text = $('<p>').html(options.text);
 63             
 64             if ( options.icon ) {
 65                 var $icon = $('<span>').addClass('ui-icon ' + options.icon);
 66                 $text.prepend($icon);
 67             }
 68 
 69             var $item = $('<div>')
 70                 .hide()
 71                 .addClass('notify-item  ui-corner-all ' + options.state)
 72                 .append($text)
 73                 .animate(options.inEffect, options.effectTime)
 74                 .click(function() {
 75                     $(this).animate(options.outEffect, options.effectTime, function(){
 76                         $(this).animate({height: '0px'}, options.effectTime, function(){
 77                             $(this).remove();
 78                         });
 79                     });
 80                 })
 81                 .hover(function(){
 82                     $(this).addClass('ui-state-disabled');
 83                 }, function(){
 84                     $(this).removeClass('ui-state-disabled');
 85                 });
 86 
 87             // Append to wrapper
 88             $this.append($item);
 89 
 90             // Autoremove
 91             if ( options.remove ) {
 92                 setTimeout(function(){
 93                     if ( $item && $item.click ) {
 94                         $item.click();
 95                     }
 96                 },
 97                 options.timeout);
 98             }
 99         });
100     }
101 
102 })(jQuery);
